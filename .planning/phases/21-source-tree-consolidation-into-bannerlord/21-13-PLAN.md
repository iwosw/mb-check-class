Suggested script skeleton (executor may adapt):

```python
import json, sys
from pathlib import Path

KEEP_PREFIXES = (
    "gui.recruits.", "key.recruits.", "category.recruits", "chat.recruits.",
    "description.recruits.", "subtitles.recruits.", "recruits.", "gui.multiLineEditBox.",
)
DROP_PREFIXES = ("entity.recruits.", "item.recruits.", "block.recruits.")

def merge(src_path: Path, tgt_path: Path) -> dict:
    src = json.loads(src_path.read_text(encoding="utf-8"))
    tgt = json.loads(tgt_path.read_text(encoding="utf-8"))
    kept = skipped_obsolete = skipped_present = inserted = 0
    for key, value in src.items():
        if any(key.startswith(p) for p in DROP_PREFIXES):
            skipped_obsolete += 1
            continue
        if not any(key.startswith(p) or key == p.rstrip(".") for p in KEEP_PREFIXES):
            continue
        kept += 1
        if key in tgt:
            skipped_present += 1
            continue
        tgt[key] = value
        inserted += 1
    tgt_path.write_text(json.dumps(tgt, indent=2, ensure_ascii=False) + "\n", encoding="utf-8")
    json.loads(tgt_path.read_text(encoding="utf-8"))  # re-parse for validation
    return {"kept": kept, "skipped_obsolete": skipped_obsolete, "skipped_already_present": skipped_present, "inserted": inserted}

PAIRS = [
    ("en_us", "recruits/src/main/resources/assets/recruits/lang/en_us.json", "src/main/resources/assets/bannermod/lang/en_us.json"),
    ("ru_ru", "recruits/src/main/resources/assets/recruits/lang/ru_ru.json", "src/main/resources/assets/bannermod/lang/ru_ru.json"),
    ("de_de", "recruits/src/main/resources/assets/recruits/lang/de_de.json", "src/main/resources/assets/bannermod/lang/de_de.json"),
    ("ja_jp", "recruits/src/main/resources/assets/recruits/lang/ja_jp.json", "src/main/resources/assets/bannermod/lang/ja_jp.json"),
    ("tr_tr", "recruits/src/main/resources/assets/recruits/lang/tr_tr.json", "src/main/resources/assets/bannermod/lang/tr_tr.json"),
    ("es_es (en_us fallback)", "recruits/src/main/resources/assets/recruits/lang/en_us.json", "src/main/resources/assets/bannermod/lang/es_es.json"),
]
for label, src, tgt in PAIRS:
    stats = merge(Path(src), Path(tgt))
    print(f"{label}: {stats}")
```
